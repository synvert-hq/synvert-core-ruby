def encode(source)
  source.gsub(/^.*?<%=?/m) { |str| ' ' * str.size }
        .gsub(/%>.*?$/m) { |str| ' ' * str.size }
        .gsub(/%>.*?<%=?/m) { |str| ' ' * str.size }
end
